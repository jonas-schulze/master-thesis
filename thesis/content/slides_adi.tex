  \begin{block}{General Formulation \parencite{Peaceman1955}}
    For $Ax=b$ with $A = H+V$ the method reads
    \begin{equation*}
    \left\{
    \begin{aligned}
      \Aip  x^{k+\frac{1}{2}} &= b - \Aiim x^k \\
      \Aiip x^{k+1}           &= b - \Aim x^{k+\frac{1}{2}}
    \end{aligned}
    \right.
    \end{equation*}
    where
    $\Aip  := H + \alpha_k I_n$,
    $\Aiip := V + \beta_k  I_n$,
    $\Aim  := H - \beta_k  I_n$,
    $\Aiim := V - \alpha_k I_n$,
    and $\alpha_k, \beta_k\in\C$.
  \end{block}

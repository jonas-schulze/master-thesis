\begin{frame}[t]{Estimating Parareal Runtime}
  \begin{block}{Runtime Model}
  %\vspace{-\baselineskip} % maybe needed if in column
  \begin{equation*}
    \hattpar
    := \twarmup
    + \underbrace{
      \strut
      N \cdot (\trampup + t_G)
    }_{\substack{
      \text{stages $1\leq n\leq N$}\\
      \text{refinement $k=0$}
    }}
    + \underbrace{
      \strut
      K \cdot (t_F + t_G)
    }_{\substack{
      \vphantom{\text{stage}}
      n = N \\
      1 \leq k \leq K
    }}
    + t_F
  \end{equation*}
  \end{block}
  \only<1>{%

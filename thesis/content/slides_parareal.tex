\subsection{Parareal Method}

\begin{frame}<-2>{Parareal Method}
\begin{bigpicturecols}
  \begin{block}{General Formulation \parencite{Lions2001}}
    For the IVP $\dot u = f(u)$ the method reads
    \begin{equation*}
      \left\{
      \begin{aligned}
        U^0_{n+1} &:= G(U^0_n) \\
        U^{k+1}_{n+1} &:= G(U^{k+1}_n) + F(U^k_n) - G(U^k_n)
      \end{aligned}
      \right.
    \end{equation*}
    where $U_0^0 := u(t_0)$. $U_n^k$ converges to $u(t_n)$ as $k\to\infty$.
  \end{block}
  \begin{itemize}
    \item
      KÃ¶hler, Saak, and Lang 2016 (GAMM):
      LRSIF formulation
  % GAMM Annual Meeting
  \begin{equation*}
    U^{k+1}_{n+1}
    =\vphantom{\Bigg[}\parbox{6cm}{$%

    + \colorldlt{\colb}
    - \colorldlt{\colc}

    \Bigg[
    \begin{matrix}

    }$}
  \end{equation*}
    \item

      where $0 \leq n \leq N$.
  \end{itemize}

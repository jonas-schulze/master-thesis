      \ifnum#2<2 \pause\fi

      \ifnum#2<3 \pause\fi

        Rosenbrock stages\\ are ALEs
      \ifnum#2<5 \pause\fi
      \item
        ADI method\\ to solve ALE
      \ifnum#2<6 \pause\fi
      \item
        Parareal method\\ for speed-up
    \end{itemize}
  \end{block}
}
